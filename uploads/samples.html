Language-specific Instructions

C

Command line: 
gcc -m32 example.c
./a.out
The -m32 flag compiles your program as 32-bit, since the judging servers are 32-bit.
If you need a 32-bit integer, use int or int32_t.
If you need a 64-but integer, use long long or int64_t 
#include <stdio.h>

int main() {
	//read integer
	int x;
	scanf("%d", &x);

	//read string (less than 1000 characters)
	char str[1000];
	scanf("%s", str);

	//read line (less than 1000 characters)
	gets(str);

	// read lines until end (tip: when testing in a terminal, Ctrl-D ends input)
	while(scanf("%s", str), !feof(stdin)) {
	}

	//output
	printf("%s", "Hello worldn");

	return 0; //always return 0, or else your program will fail
}

C++

Command line: 
g++ -m32 example.c
./a.out
The -m32 flag compiles your program as 32-bit, since the judging servers are 32-bit.
If you need a 32-bit integer, use int or int32_t.
If you need a 64-but integer, use long long or int64_t 
#include <iostream>
#include <string>

int main() {
	//read integer
	int x;
	cin >> x;

	//read string
	string str;
	cin >> str;

	//read line
	getline(cin, str);

	//read lines until end (tip: when testing in a terminal, Ctrl-D ends input)
	while(cin >> str) {
	}

	//output
	cout << "Hello worldn";
}

Java

Command line: 
javac Main.java
java Main
Your source file should have no package and the class name must be Main.
If you choose to use a package or have a different class name, don't forget to change it back before you submit. 
import java.util.Scanner;

class Main {
	public static void main(String[] args) {
		Scanner input = new Scanner(System.in);

		//read integer
		int x = input.nextInt();

		//read string
		String str = input.next();

		//read line
		//FYI, generally you should use either nextLine or next/nextInt/...,
		//but not both
		str = input.nextLine();

		//read strings until end (tip: when testing in a terminal, Ctrl-D ends input)
		while(input.hasNext()) {
			str = input.next();
		}

		//output
		System.out.println("Hello world");
	}
}

Node.js

Command line: 
node example.js
/**
 * The simplest way is to read in all input first, and then operate on that.
 * On the command line, use Ctrl-D to end input.
 */

process.stdin.resume();
process.stdin.setEncoding('utf8');
var chunks = [];
process.stdin.on('data', function (chunk) {
	chunks.push(chunk);
});
process.stdin.once('end', function() {
	main(chunks.join(''));
});

function main(input) {
	//do something with input, like input.split('n')

	//output
	console.log('Hello world'); //or process.stdout.write('Hello world');
}
